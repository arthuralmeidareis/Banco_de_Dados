--1
select distinct nome_medico
from medico m, consulta c, exame e, lista_exame l
WHERE m.cod_medico = c.cod_medico
and c.cod_consulta = l.cod_consulta
and l.cod_exame = e.cod_exame
and valor_exame = (select max(valor_exame) from exame);
--2
select distinct nome_paciente
from consulta c, paciente p , exame e, lista_exame l 
where c.cod_paciente = p.cod_paciente
and c.cod_consulta = l.cod_consulta
and l.cod_exame = e.cod_exame
and valor_exame = (select max (valor_exame)  from exame);
--7
select distinct nome_medico,data_consulta
from medico m , consulta c , lista_exame l , exame e
where m.cod_medico = c.cod_medico
and l.cod_consulta = c.cod_consulta
and l.cod_exame = e.cod_exame
and data_consulta = (select min(data_consulta) from consulta);
--9
select distinct nome_medico, data_consulta
from medico m , consulta c 
where m.cod_medico = c.cod_medico
and to_char (data_admissao,'DD/MM/YY') = (select (to_char( max(data_admissao),'DD/MM/YY')) from medico);
--10
select distinct  nome_medico,  nome_paciente, idade_paciente
from medico m , consulta c, paciente p
where m.cod_medico = c.cod_medico
and p.cod_paciente = c.cod_paciente
and idade_paciente = (select max (idade_paciente) from paciente);

--11
select distinct nome_paciente, data_consulta
from paciente p , consulta c
where p.cod_paciente = c.cod_paciente
and data_consulta = (select max (data_consulta) from consulta);
